事件处理

  • XEvent是所有事件结构的联合类型，因此需要类型转换

    XNextEvent(display,rep) /* flush output then fill in struct with event */

    XSelectInput(display,w, mask)/* declare which events you wish to receive */

    XFlush(display) /* flush all output buffers */

    XSync(display,discard) /* flush output buffers & wait till all events have been　delivered. If discard <> 0 throw all the events away. */

    XPeekEvent(display,rep) /* flush output then fill in struct with event, RETURN next event but DONT remove from queue if queue is empty it waits */

    int XQLength(display) /* returns the length of the waiting event queue */

    XPutBackEvent(display,event) /* push an event back onto the top of the queue */

    XWindowEvent(display,w, mask, rep) /* flush then look down the queue for match*/

    Bool XCheckWindowEvent(display,w, mask, rep) /* async */

    XMaskEvent(display,mask, rep)/* sync for all input */

    Bool XCheckMaskEvent(display,mask, rep)/* async for all input */

    int XPending(display) /* flush output & return no of events in input queue */
